```typescript
// logs-service/src/validations/logs.ts
export const LogFiltersSchema = z.object({
  service: z.string().optional(),
  level: LogLevelSchema.optional(),
  environment: z.string().optional(),
  userId: z.string().optional(),
  requestId: z.string().optional(),
  sessionId: z.string().optional(),

  endDate: z.coerce.date().optional(),
  limit: z.coerce.number().min(1).max(1000).default(100),
  offset: z.coerce.number().min(0).default(0),
});
```

```typescript
// logs-service/src/services/logs.service.ts
export async function getLogs(
  filters: LogFiltersRequest,
): Promise<LogsWithPagination> {
  const client = await getClient();
  const collection = client.db().collection("logs");

  // Construction du filtre MongoDB
  const mongoFilter: any = {};

  if (filters.service) mongoFilter.service = filters.service;
  if (filters.level) mongoFilter.level = filters.level;
  if (filters.environment) mongoFilter.environment = filters.environment;
  if (filters.userId) mongoFilter.userId = filters.userId;
  if (filters.requestId) mongoFilter.requestId = filters.requestId;
  if (filters.sessionId) mongoFilter.sessionId = filters.sessionId;

  // Filtres de date
  if (filters.startDate || filters.endDate) {
    mongoFilter.timestamp = {};
    if (filters.startDate) mongoFilter.timestamp.$gte = filters.startDate;
    if (filters.endDate) mongoFilter.timestamp.$lte = filters.endDate;
  }

  // Exécution des requêtes en parallèle
  const [logs, total] = await Promise.all([
    collection
      .find<LogDocument>(mongoFilter)
      .sort({ timestamp: -1 })
      .skip(filters.offset)
      .limit(filters.limit)
      .toArray(),
    collection.countDocuments(mongoFilter),
  ]);

  return {
    logs,
    total,
  };
}
```

```typescript
// logs-service/src/controllers/logs.controller.ts
export async function getLogs(req: Request, res: Response): Promise<void> {
  // Validation avec Zod
  const filters = LogFiltersSchema.parse(req.query);

  // Utilisation du service MongoDB
  const { logs, total } = await logService.getLogs(filters);

  res.json({
    data: logs,
    pagination: {
      total,
      limit: filters.limit,
      offset: filters.offset,
    },
  });
}
```

```typescript
export async function getLogById(id: string): Promise<LogDocument | null> {
  const client = await getClient();
  const collection = client.db().collection("logs");

  const log = await collection.findOne<LogDocument>({ _id: new ObjectId(id) });

  if (!log) return null;

  return log;
}
```

```typescript
// logs-service/src/services/logs.service.ts
export async function getLogStats(
  filters: LogStatsFiltersRequest,
): Promise<LogStats> {
  const client = await getClient();
  const collection = client.db().collection("logs");

  // Construction du filtre MongoDB
  const mongoFilter: any = {};

  if (filters.service) mongoFilter.service = filters.service;
  if (filters.environment) mongoFilter.environment = filters.environment;

  if (filters.startDate || filters.endDate) {
    mongoFilter.timestamp = {};
    if (filters.startDate) mongoFilter.timestamp.$gte = filters.startDate;
    if (filters.endDate) mongoFilter.timestamp.$lte = filters.endDate;
  }

  // Exécution des agrégations en parallèle
  const [totalLogs, logsByLevel, logsByService, logsByEnvironment] =
    await Promise.all([
      // Nombre total de logs
      collection.countDocuments(mongoFilter),

      // Répartition par niveau
      collection
        .aggregate([
          { $match: mongoFilter },
          { $group: { _id: "$level", count: { $sum: 1 } } },
        ])
        .toArray(),

      // Répartition par service (top 20)
      collection
        .aggregate([
          { $match: mongoFilter },
          { $group: { _id: "$service", count: { $sum: 1 } } },
          { $sort: { count: -1 } },
          { $limit: 20 },
        ])
        .toArray(),

      // Répartition par environnement
      collection
        .aggregate([
          { $match: mongoFilter },
          { $group: { _id: "$environment", count: { $sum: 1 } } },
        ])
        .toArray(),
    ]);

  // Transformation des résultats en format plus lisible
  const stats: LogStats = {
    totalLogs,
    logsByLevel: logsByLevel.reduce(
      (acc, item) => {
        acc[item._id] = item.count;
        return acc;
      },
      {} as Record<string, number>,
    ),
    logsByService: logsByService.reduce(
      (acc, item) => {
        acc[item._id] = item.count;
        return acc;
      },
      {} as Record<string, number>,
    ),
    logsByEnvironment: logsByEnvironment.reduce(
      (acc, item) => {
        acc[item._id] = item.count;
        return acc;
      },
      {} as Record<string, number>,
    ),
  };

  return stats;
}
```

## Création du microservice files-service

### Initialisation du projet

Créer un dossier `files-service` + `npm init` + installation des dépendances comme pour le log service sauf mongodb

### Fichiers nécessaires

- `index.ts` : point d'entrée du serveur
- `src/app.ts` : configuration de l'application Express
- `src/routes/index.ts` : fichier de routage principal
- `src/routes/files.routes.ts` : routes pour les fichiers
- `src/controllers/file.controller.ts` : contrôleur pour les fichiers
- `src/middlewares/*` : middlewares (recopier les middlewares de logs-service)

Ne pas oublier de renseigner les middlewares dans le fichier `src/app.ts`

```typescript
// files-service/src/lib/upload.ts
export const uploadDir = path.join(import.meta.dirname, "../../uploads");
export const formParser = formidable({
  uploadDir,
  createDirsFromUploads: true,
  keepExtensions: true,
  // SI on souhaite customiser le nom du fichier / le rangement du fichier
  filename(name, ext, part, form) {
    const currentDate = new Date();
    // Generate a unique filename based on the current date and original name
    const newName = crypto.randomBytes(16).toString("hex");
    return `${currentDate.getFullYear()}/${currentDate.getMonth() + 1}/${currentDate.getDate()}/${newName}${ext}`;
  },
});
```

```typescript
// files-service/src/controllers/file.controller.ts
export async function uploadFile(req: Request, res: Response): Promise<void> {
  const [fields, files] = await formParser.parse(req);

  // files est un objet avec des propriétés contenant des tableaux de fichiers uploadés
  // ex: { file: [{...}], image: [{...}] }

  // A manipuler, cela peut être contraignant car on doit savoir quelle propriété est utilisée pour l'upload
  // Moi je ne souhaite que récupérer les tableaux de fichiers uploadés
  // je souhaite passer de { file: [{...}], image: [{...}] } à un tableau de fichiers [{...}, {...}]

  // 1ère transformation : Object.entries(files)
  // { file: [{...}], image: [{...}] } => [['file', [{...}]], ['image', [{...}]]]
  const filesToSave = Object.entries(files)
    // 2ème transformation : on récupère les valeurs de chaque entrée
    // [['file', [{...}]], ['image', [{...}]]] => [[{...}], [{...}]]
    .map(([key, files]) => files || [])
    // 3ème transformation : on a un tableau de tableaux, on veut un tableau à plat
    // [[{...}], [{...}]] => [{...}, {...}]
    .flat();

  res.json(filesToSave);
}
```
