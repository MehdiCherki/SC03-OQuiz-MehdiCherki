### Services

Quand on aura besoin des types pour la création de log, faire un fichier `src/validations/logs.ts` et faire le schéma de validation (voir [logs.ts](../../logs-service/src/validations/logs.ts))

```typescript
// src/services/logs.service.ts
import { getClient } from '../lib/db.ts';
import { ObjectId } from 'mongodb';
import { type CreateLogRequest } from '../validations/logs.ts';

export interface LogDocument {
  _id: ObjectId;
  timestamp: Date;
  level: LogLevel;
  message: string;
  service: string;
  version?: string;
  environment: string;
  userId?: string;
  requestId?: string;
  sessionId?: string;
  hostname?: string;
  ip?: string;
  userAgent?: string;
  metadata?: any;
  stackTrace?: string;
}

export async function createLog(data: CreateLogRequest) {
  const logData = {
    ...data,
    timestamp: new Date(),
    environment: data.environment || 'development',
  };

  const client = await getClient();
  await client.db().collection('logs').insertOne(logData);
}

export async function getLogById(id: string): Promise<LogDocument | null> {
  const client = await getClient();
  const collection = client.db().collection('logs');

  const log = await collection.findOne<LogDocument>({ _id: new ObjectId(id) });

  if (!log) return null;

  return log;
}

export async function getLogs(): Promise<LogDocument[]> {
  const client = await getClient();
  const collection = client.db().collection('logs');

  const logs = await collection.find<LogDocument>().sort({ timestamp: -1 }).toArray();

  return logs;
}
```

### Route pour la création de log

```typescript
// src/controllers/logs.controller.ts
import type { Request, Response } from 'express';
import { CreateLogSchema } from '../validations/logs.ts';
import * as logService from '../services/logs.service.ts';

export async function createLog(req: Request, res: Response): Promise<void> {
  // Validation avec Zod (lance automatiquement ZodError si invalide)
  const logData = CreateLogSchema.parse(req.body);

  await logService.createLog(logData);

  res.status(201).json({
    message: 'Log créé avec succès',
  });
}
```

```typescript
// src/routers/log.router.ts
import { Router } from 'express';
import { createLog } from '../controllers/logs.controller.ts';

export const router = Router();

// Création de logs
router.post('/logs', createLog);
```

### On branche notre logger de oqcm à notre service

On ajoute un transport HTTP pour envoyer les logs au service de logs

```typescript
// api/config.ts
export const config = {
  // ...
  logsServiceUrl: process.env.LOGS_SERVICE_HOST || 'http://localhost',
  logsServicePort: process.env.LOGS_SERVICE_PORT || 3030,
};
```

```typescript
// api/src/lib/logger.ts
// ...
new transports.Http({
  host: config.logsServiceHost,
  port: config.logsServicePort,
  path: '/logs',
  level: 'http',
});
```

On utilise une application pour accéder à la BDD de log (TablePlus / mongo compass / ...) et voir les logs s'ajouter dans la BDD.
